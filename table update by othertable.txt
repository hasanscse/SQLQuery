UPDATE 
    PRO 
SET 
    PRO.suplier = BUYER_SUPPLIER_INFO.BUYER_SUPPLIER_ID
   
  
FROM 
    BUYER_SUPPLIER_INFO 
   -- [INNER | LEFT] JOIN t2 ON join_predicate
WHERE 
    PRO.suplier=BUYER_SUPPLIER_INFO.BUYER_SUPPLIER_NAME
-----------

UPDATE 
    PRODUCT 
SET 
    PRODUCT.STOCK_QUANTITY = EXCEL_PRODUCT.Total_Quantity,
    PRODUCT.SALES_PRICE=EXCEL_PRODUCT.Unit_Sell_Price,
    PRODUCT.PURCHASE_PRICE=EXCEL_PRODUCT.Unit_Purchase_Price,
    PRODUCT.PURCHASED_PRICE=EXCEL_PRODUCT.Unit_Purchase_Price,
    PRODUCT.TOTAL_STOCK_AMOUNT=(EXCEL_PRODUCT.Unit_Purchase_Price*EXCEL_PRODUCT.Total_Quantity)
    
    
  
FROM 
    EXCEL_PRODUCT 
   -- [INNER | LEFT] JOIN t2 ON join_predicate
WHERE 
    PRODUCT.PRODUCT_CODE=EXCEL_PRODUCT.Product_Code
-----------------------------

UPDATE 
    PRO 
SET 
    PRO.cat = PRODUCT_CATEGORY.CATEGORY_ID
   
  
FROM 
    PRODUCT_CATEGORY 
   -- [INNER | LEFT] JOIN t2 ON join_predicate
WHERE 
    PRO.cat=PRODUCT_CATEGORY.CATEGORY_NAME


--------------

UPDATE 
    SALES_DETAILS 
SET 
    
    SALES_DETAILS.PURCHASE_PRICE=PRODUCT.PURCHASE_PRICE
    
    
    
  
FROM 
    PRODUCT 
   -- [INNER | LEFT] JOIN t2 ON join_predicate
WHERE 
    SALES_DETAILS.PRODUCT_CODE=PRODUCT.PRODUCT_CODE
