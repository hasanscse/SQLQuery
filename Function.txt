
ALTER FUNCTION [dbo].[GET_SERIAL_LIST_FOR_PURCHASE_BACK]
(@PRODUCT_CODE NVARCHAR(30),@COMPANY_ID NVARCHAR(30),@PURCHASE_INVOICE_NO NVARCHAR(30),@EDIT_NUMBER int)
RETURNS NVARCHAR(2000)
AS
BEGIN
DECLARE @PRODUCT_SERIAL NVARCHAR(1000)
SELECT @PRODUCT_SERIAL=COALESCE(@PRODUCT_SERIAL+',  ','')+S.PRODUCT_SERIAL
 FROM PURCHASED_PRODUCT_SERIAL_BACK S
 WHERE S.PRODUCT_CODE = @PRODUCT_CODE AND 
 S.PURCHASE_INVOICE_NO=@PURCHASE_INVOICE_NO  and S.COMPANY_ID=@COMPANY_ID and s.EDIT_NUMBER = @EDIT_NUMBER
RETURN @PRODUCT_SERIAL
END
